308{

310 void* m = malloc(UNITY_INTERNAL_HEAP_SIZE_BYTES/2);
311 void* n1 = malloc(10);

314

320 TEST_ASSERT_NULL(out_of_mem); /* The realloc should have failed */

322 TEST_ASSERT_MEMORY_ALL_FREE_LIFO_ORDER(m, n2);
