30int UnityMain(int argc, const char* argv[], void (*runAllTests)(void))
31{
32 int result = UnityGetCommandLineOptions(argc, argv);
33 unsigned int r;
34 if (result != 0)
35 return result;
36
37 for (r = 0; r < UnityFixture.RepeatCount; r++)
38 {
39 UnityBegin(argv[0]);
40 announceTestRun(r);
41 runAllTests();
42 if (!UnityFixture.Verbose) UNITY_PRINT_EOL();
43 UnityEnd();
44 }
45
46 return (int)Unity.TestFailures;
47}

70void UnityTestRunner(unityfunction* setup,
71 unityfunction* testBody,
72 unityfunction* teardown,
73 const char* printableName,
74 const char* group,
75 const char* name,
76 const char* file,
77 unsigned int line)
78{
79 if (testSelected(name) && groupSelected(group))
80 {
81 Unity.TestFile = file;
82 Unity.CurrentTestName = printableName;
83 Unity.CurrentTestLineNumber = line;
84 if (UnityFixture.Verbose)
85 {
86 UnityPrint(printableName);
87 #ifndef UNITY_REPEAT_TEST_NAME
88 Unity.CurrentTestName = NULL;
89 #endif
90 }
91 else if (UnityFixture.Silent)
92 {
93 /* Do Nothing */
94 }
95 else
96 {
97 UNITY_OUTPUT_CHAR('.');
98 }
99
100 Unity.NumberOfTests++;
101 UnityPointer_Init();
102
103 if (!UnityFixture.DryRun) {
104 UNITY_EXEC_TIME_START();
105
106 if (TEST_PROTECT())
107 {
108 setup();
109 testBody();
110 }
111 if (TEST_PROTECT())
112 {
113 teardown();
114 }
115 }
116
117 if (TEST_PROTECT())
118 {
119 UnityPointer_UndoAllSets();
120 }
121 UnityConcludeFixtureTest();
122 }
123}

163void UnityPointer_Set(void** pointer, void* newValue, UNITY_LINE_TYPE line)
164{
165 if (pointer_index >= UNITY_MAX_POINTERS)
166 {
167 UNITY_TEST_FAIL(line, "Too many pointers set");
168 }
169 else
170 {
171 pointer_store[pointer_index].pointer = pointer;
172 pointer_store[pointer_index].old_value = *pointer;
173 *pointer = newValue;
174 pointer_index++;
175 }
176}
177
178void UnityPointer_UndoAllSets(void)
179{
180 while (pointer_index > 0)
181 {
182 pointer_index--;
183 *(pointer_store[pointer_index].pointer) =
184 pointer_store[pointer_index].old_value;
185 }
186}
