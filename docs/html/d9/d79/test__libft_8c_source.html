16void check_memory_leaks(void) {
17 printf("Starting memory leak check...\n");
18 #ifdef __APPLE__
19 printf("Running leaks command...\n");
20 int result = system("leaks test_libft > /dev/null");
21 if (result != 0) {

24 printf("No memory leaks detected.\n");

26 #elif defined(__linux__)
27 printf("Running valgrind...\n");
28 int result = system("valgrind --leak-check=full --show-leak-kinds=all --track-origins=yes --verbose --log-file=valgrind-out.txt ./test_libft > /dev/null");
29 if (result != 0) {
30 printf("Valgrind command failed with status %d\n", result);
31 return;
32 }
33
34 FILE *fp = fopen("valgrind-out.txt", "r");
35 if (!fp) {
36 printf("Failed to open valgrind output file\n");
37 return;
38 }
39
40 char line[256];
41 int leaks_found = 0;
42 while (fgets(line, sizeof(line), fp)) {
43 if (strstr(line, "definitely lost:") ||
44 strstr(line, "indirectly lost:") ||
45 strstr(line, "possibly lost:")) {
46 if (strstr(line, "0 bytes")) continue;
47 leaks_found = 1;
48 printf("Memory Leak Detected: %s", line);
49 }
50 }
51 fclose(fp);
52
53 if (!leaks_found) {
54 printf("No memory leaks detected.\n");
55 }
56
57 if (unlink("valgrind-out.txt") != 0) {
58 printf("Warning: Failed to remove valgrind output file\n");
59 }

65double measure_time(void (*test_func)(void)) {
66 struct timeval start, end;
67 gettimeofday(&start, NULL);
68 test_func();
69 gettimeofday(&end, NULL);
70 return (end.tv_sec - start.tv_sec) + (end.tv_usec - start.tv_usec) / 1000000.0;
