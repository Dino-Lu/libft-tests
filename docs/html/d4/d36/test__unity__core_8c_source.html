292{
293#ifndef USING_OUTPUT_SPY
294 TEST_IGNORE();
295#else
296 int failures = 0;
297 UNITY_COUNTER_TYPE savedGetFlushSpyCalls = 0;
298 UNITY_COUNTER_TYPE savedFailures = Unity.TestFailures;

304 endPutcharSpy();
305 TEST_ASSERT_EQUAL(0, savedGetFlushSpyCalls);
306 TEST_ASSERT_EQUAL(savedFailures + 1, Unity.TestFailures);
307#if defined(UNITY_OUTPUT_FLUSH) && defined(UNITY_OUTPUT_FLUSH_HEADER_DECLARATION)
308 TEST_ASSERT_EQUAL(1, getFlushSpyCalls());
309#else
310 TEST_ASSERT_EQUAL(0, getFlushSpyCalls());
311#endif
312 endFlushSpy();
313
314 startPutcharSpy(); /* Suppress output */
315 failures = UnityEnd();
316 Unity.TestFailures--;
317 endPutcharSpy();
318 TEST_ASSERT_EQUAL(savedFailures + 1, failures);
319#endif
