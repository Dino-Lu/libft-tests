68void* unity_malloc(size_t size)
69{
70 char* mem;
71 Guard* guard;
72 size_t total_size;
73
74 total_size = sizeof(Guard) + unity_size_round_up(size + sizeof(end));
75
76 if (malloc_fail_countdown != MALLOC_DONT_FAIL)
77 {
78 if (malloc_fail_countdown == 0)
79 return NULL;
80 malloc_fail_countdown--;
81 }
82
83 if (size == 0) return NULL;
84#ifdef UNITY_EXCLUDE_STDLIB_MALLOC
85 if (heap_index + total_size > UNITY_INTERNAL_HEAP_SIZE_BYTES)
86 {
87 guard = NULL;
88 }
89 else
90 {
91 /* We know we can get away with this cast because we aligned memory already */
92 guard = (Guard*)(void*)(&unity_heap[heap_index]);
93 heap_index += total_size;
94 }
95#else
96 guard = (Guard*)UNITY_MALLOC(total_size);

98 if (guard == NULL) return NULL;
99 malloc_count++;
100 guard->size = size;
101 guard->guard_space = 0;
102 mem = (char*)&(guard[1]);
103 memcpy(&mem[size], end, sizeof(end));
104

164void* unity_realloc(void* oldMem, size_t size)
165{
166 Guard* guard = (Guard*)oldMem;
167 void* newMem;
168
169 if (oldMem == NULL) return unity_malloc(size);
170
171 guard--;
172 if (isOverrun(oldMem))
173 {
174 release_memory(oldMem);
175 UNITY_TEST_FAIL(Unity.CurrentTestLineNumber, "Buffer overrun detected during realloc()");
176 }
177
178 if (size == 0)
179 {
180 release_memory(oldMem);
181 return NULL;
182 }
183
184 if (guard->size >= size) return oldMem;
185
186#ifdef UNITY_EXCLUDE_STDLIB_MALLOC /* Optimization if memory is expandable */
187 {
188 size_t old_total_size = unity_size_round_up(guard->size + sizeof(end));
189
190 if ((oldMem == unity_heap + heap_index - old_total_size) &&
191 ((heap_index - old_total_size + unity_size_round_up(size + sizeof(end))) <= UNITY_INTERNAL_HEAP_SIZE_BYTES))
192 {
193 release_memory(oldMem); /* Not thread-safe, like unity_heap generally */
194 return unity_malloc(size); /* No memcpy since data is in place */
195 }
196 }
197#endif
198 newMem = unity_malloc(size);
199 if (newMem == NULL) return NULL; /* Do not release old memory */
200 memcpy(newMem, oldMem, guard->size);
201 release_memory(oldMem);
202 return newMem;
203}
